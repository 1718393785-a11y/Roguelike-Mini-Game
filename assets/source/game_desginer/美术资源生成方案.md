  - 边缘清晰，无模糊抗锯齿

## 5. 图片识别与映射方案

### 方案A: 文件命名约定自动识别

```javascript
// 【命名规范】 asset_{类型}_{ID}_{等级}.png
// 示例:
//   asset_weapon_saber_6.png    → 百炼环首刀 Lv.6
//   asset_enemy_soldier.png     → 魏兵
//   asset_skill_DAMAGE.png      → 陷阵杀气被动
//   asset_ui_hud.png            → HUD界面

const assetFileRegex = /asset_(\w+)_(\w+)(?:_(\d+))?\.png/;

function parseAssetFilename(filename) {
    const match = filename.match(assetFileRegex);
    if (!match) return null;
    
    const [, type, id, level] = match;
    return {
        type,           // weapon, enemy, skill, ui, pickup
        id,             // saber, soldier, DAMAGE, hud
        level: level ? parseInt(level) : null,
        filepath: `assets/${type}s/${filename}`
    };
}

// 自动扫描assets目录并建立映射表
function buildAssetMapping() {
    const mapping = { weapons: {}, enemies: {}, skills: {}, ui: {}, pickups: {} };
    
    // 扫描并分类所有资源文件
    for (const file of scanAssetDirectory()) {
        const parsed = parseAssetFilename(file);
        if (!parsed) continue;
        
        const category = `${parsed.type}s`;
        if (!mapping[category][parsed.id]) {
            mapping[category][parsed.id] = {};
        }
        
        if (parsed.level) {
            mapping[category][parsed.id][parsed.level] = parsed.filepath;
        } else {
            mapping[category][parsed.id] = parsed.filepath;
        }
    }
    
    return mapping;
}
```

### 方案C: 运行时动态加载器

```javascript
// src/render/AssetLoader.ts
class AssetLoader {
    private mapping: AssetMapping;
    private cache: Map<string, HTMLImageElement>;
    
    async initialize() {
        // 加载映射表
        const response = await fetch('assets/asset-mapping.json');
        this.mapping = await response.json();
        this.cache = new Map();
    }
    
    // 获取武器图标
    getWeaponSprite(weaponId: string, level: number): HTMLImageElement | null {
        const path = this.mapping.weapons[weaponId]?.[level];
        return path ? this.loadAndCache(path) : null;
    }
    
    // 获取敌人精灵
    getEnemySprite(enemyId: string): HTMLImageElement | null {
        const path = this.mapping.enemies[enemyId];
        return path ? this.loadAndCache(path) : null;
    }
    
    // 获取被动技能图标
    getSkillIcon(skillId: string): HTMLImageElement | null {
        const path = this.mapping.skills[skillId];
        return path ? this.loadAndCache(path) : null;
    }
    
    // 自动根据配置选择资源
    autoSelectWeaponSprite(weapon: Weapon): HTMLImageElement {
        const spec = WEAPON_SPEC[weapon.id];
        const effectiveLevel = Math.min(weapon.level, 6);
        
        // 优先找对应等级图标，降级到 Lv.1，再降级为占位
        return this.getWeaponSprite(weapon.id, effectiveLevel)
            || this.getWeaponSprite(weapon.id, 1)
            || this.getFallbackIcon();
    }
    
    private loadAndCache(path: string): HTMLImageElement {
        if (this.cache.has(path)) {
            return this.cache.get(path)!;
        }
        
        const img = new Image();
        img.src = path;
        this.cache.set(path, img);
        return img;
    }
}
```

// tools/validate-assets.js

// 检查资源完整性
function validateAssetCompleteness() {
    const mapping = require('../assets/asset-mapping.json');
    const issues = [];
    
    // 检查6把武器各6级图标
    for (const [weaponId, levels] of Object.entries(mapping.weapons)) {
        for (let l = 1; l <= 6; l++) {
            if (!levels[l]) {
                issues.push(`MISSING: ${weaponId} level ${l} icon`);
            }
            if (!fs.existsSync(levels[l])) {
                issues.push(`FILE_NOT_FOUND: ${levels[l]}`);
            }
        }
    }
    
    // 检查敌人
    for (const [enemyId, path] of Object.entries(mapping.enemies)) {
        if (!fs.existsSync(path)) {
            issues.push(`ENEMY_MISSING: ${enemyId} at ${path}`);
        }
    }
    
    return {
        complete: issues.length === 0,
        issues,
        stats: {
            weaponsTotal: 6 * 6,
            weaponsFound: countFoundWeapons(mapping),
            enemiesTotal: 7 + 5,
            enemiesFound: countFoundEnemies(mapping)
        }
    };
}
