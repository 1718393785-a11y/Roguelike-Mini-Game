### P0-1：全局受击无敌计时器

| 项 | 详情 |
|----|------|
| **问题** | 玩家受到连续伤害时直接被秒，没有操作空间 |
| **需求** | 玩家受伤后获得 0.5 秒无敌时间，期间闪烁 |
| **修改位置** | `Player` 类构造函数 + `takeDamage` 方法 + 渲染逻辑 |

#### 代码示例：
```javascript
// ===== Player 构造函数中新增 =====
this.invincibleTimer = 0;        // 无敌剩余时间
this.invincibleFlashSpeed = 10;  // 闪烁频率（次/秒）

// ===== 新增受击处理方法 =====
takeDamage(amount) {
    // 无敌中不受伤
    if (this.invincibleTimer > 0) return;
    
    // 计算减伤后的实际伤害
    const actualDamage = amount * (1 - this.getDamageReduction());
    
    // 扣血
    this.hp -= actualDamage;
    
    // 触发无敌
    this.invincibleTimer = 0.5;  // 0.5秒无敌
    
    // 死亡判定
    if (this.hp <= 0) {
        this.hp = 0;
        gameManager.triggerGameOver();
    }
}

// ===== Player.update 中新增计时器 =====
update(deltaTime) {
    // 无敌倒计时
    if (this.invincibleTimer > 0) {
        this.invincibleTimer -= deltaTime;
    }
    // ... 其他原有逻辑
}

// ===== Player.render 中新增闪烁逻辑 =====
render(ctx, camera) {
    // 无敌时闪烁：每 0.1 秒切换显示/隐藏
    if (this.invincibleTimer > 0) {
        const flash = Math.floor(this.invincibleTimer * this.invincibleFlashSpeed) % 2;
        if (flash === 1) {
            ctx.globalAlpha = 0.5;  // 半透明闪烁
        }
    }
    
    // ... 原有渲染逻辑 ...
    
    ctx.globalAlpha = 1;  // 恢复透明度
}
```

#### 验收标准：
- ✅ 玩家受到伤害后，0.5 秒内不会再受到第二次伤害
- ✅ 无敌期间玩家半透明闪烁
- ✅ 多只怪物同时攻击时，只会受到第一只的伤害
- ✅ 无敌结束后恢复正常受击

### P0-2：玩家受击推开周围怪物

| 项 | 详情 |
|----|------|
| **问题** | 怪物贴脸后玩家没有任何反制手段 |
| **需求** | 玩家受伤时，反向推开周围 80 范围内的所有小怪 |
| **修改位置** | `Player.takeDamage` 方法中调用 |

#### 代码示例：
```javascript
// 在 Player.takeDamage 中扣血后加：
takeDamage(amount) {
    // ... 原有扣血逻辑 ...
    
    // 受击后推开周围怪物，创造输出空间
    this.knockbackEnemiesAround();
}

// 新增方法：推开周围敌人
knockbackEnemiesAround() {
    const radius = 80;       // 推开范围
    const force = 30;        // 推开力度
    const gm = window.gameManager;
    
    for (let enemy of gm.enemies) {
        // Boss/木牛完全免疫击退
        if (enemy.knockbackResist >= 1.0) continue;
        
        const dx = enemy.x - this.x;
        const dy = enemy.y - this.y;
        const dist = Math.sqrt(dx * dx + dy * dy);
        
        if (dist < radius && dist > 0) {
            // 距离越近推得越远
            const normalized = 1 - (dist / radius);
            const actualForce = force * normalized * (1 - enemy.knockbackResist);
            
            enemy.x += (dx / dist) * actualForce;
            enemy.y += (dy / dist) * actualForce;
        }
    }
}
```

#### 验收标准：
- ✅ 玩家受伤时，周围普通怪物会被向外弹开
- ✅ 距离越近弹得越远
- ✅ Boss、木牛完全免疫（不被推开）
- ✅ 精英怪击退效果减半（符合 50% 抗性设定）

### P1-2：Boss词缀系统扩展

| 项 | 详情 |
|----|------|
| **问题** | 目前5关Boss只有固定能力，缺少变化和重复可玩性 |
| **需求** | 第3关及以后的Boss随机获得1-2个额外词缀 |
| **修改位置** | `Boss` 类构造函数 + `Boss.update` |

#### 4种标准词缀配置：
| 词缀 | 名称 | 效果 |
|-----|------|------|
| `fearless` | **【无畏】** | Boss周期性向玩家冲锋撞击 |
| `strongbow` | **【强弓】** | Boss周期性射出8方向弧形弹幕 |
| `scorched` | **【焦土】** | 地面周期性生成火圈，持续伤害+减速 |
| `callToArms` | **【号令】** | 周期性在玩家周围召唤一圈怪物 |

#### 代码示例：
```javascript
// ===== Boss 构造函数中新增 =====
constructor(x, y, stageIndex) {
    // ... 原有逻辑 ...
    
    // 词缀系统：第3关及以后随机 1-2 个词缀
    this.affixes = [];
    if (stageIndex >= 2) {
        const available = ['fearless', 'strongbow', 'scorched', 'callToArms'];
        const count = stageIndex >= 3 ? 2 : 1;  // 第4、5关 2个词缀
        
        for (let i = 0; i < count; i++) {
            const idx = Math.floor(Math.random() * available.length);
            this.affixes.push(available[idx]);
            available.splice(idx, 1);  // 不重复
        }
    }
    
    // 词缀计时器
    this.affixTimers = {};
    for (let affix of this.affixes) {
        this.affixTimers[affix] = 3 + Math.random() * 2;  // 3-5秒后开始第一次
    }
    
    // 火圈列表（焦土词缀用）
    this.fireZones = [];
}

// ===== Boss.update 中新增词缀逻辑 =====
update(deltaTime, player) {
    // ... 原有逻辑 ...
    
    // 处理各词缀效果
    for (let affix of this.affixes) {
        this.affixTimers[affix] -= deltaTime;
        
        if (this.affixTimers[affix] <= 0) {
            this.triggerAffix(affix, player);
            this.affixTimers[affix] = this.getAffixCooldown(affix);
        }
    }
}

// 触发词缀效果
triggerAffix(affix, player) {
    const gm = window.gameManager;
    
    switch (affix) {
        case 'fearless':
            // 无畏：向玩家冲锋
            this.startCharge(player);
            break;
            
        case 'strongbow':
            // 强弓：8方向弹幕
            for (let i = 0; i < 8; i++) {
                const angle = (Math.PI * 2 / 8) * i;
                gm.enemyProjectiles.push(new Projectile(
                    this.x, this.y,
                    Math.cos(angle) * 200,
                    Math.sin(angle) * 200,
                    15, true  // 敌方投射物
                ));
            }
            break;
            
        case 'scorched':
            // 焦土：在玩家附近生成火圈
            const offsetX = (Math.random() - 0.5) * 200;
            const offsetY = (Math.random() - 0.5) * 200;
            this.fireZones.push({
                x: player.x + offsetX,
                y: player.y + offsetY,
                radius: 80,
                lifetime: 8,
                damage: 10
            });
            break;
            
        case 'callToArms':
            // 号令：在玩家周围召唤一圈怪物
            for (let i = 0; i < 8; i++) {
                const angle = (Math.PI * 2 / 8) * i;
                const spawnDist = 150;
                gm.spawnEnemyAt(
                    player.x + Math.cos(angle) * spawnDist,
                    player.y + Math.sin(angle) * spawnDist
                );
            }
            break;
    }
}

// ===== Boss.render 中新增词缀名称显示 =====
render(ctx, camera) {
    // ... 原有渲染 ...
    
    // 显示词缀名称
    if (this.affixes.length > 0) {
        ctx.fillStyle = '#ffff00';
        ctx.font = 'bold 14px sans-serif';
        ctx.textAlign = 'center';
        
        const affixNames = this.affixes.map(a => this.getAffixName(a));
        ctx.fillText(affixNames.join(' '), this.x, this.y - this.size - 20);
    }
}
```

#### 验收标准：
- ✅ 第1、2关Boss没有额外词缀
- ✅ 第3关Boss随机获得 1 个词缀
- ✅ 第4、5关Boss随机获得 2 个词缀
- ✅ Boss头顶显示词缀名称标识
- ✅ 词缀技能正确触发

### P2-2：武器冷却可视化 HUD

| 项 | 详情 |
|----|------|
| **问题** | 玩家不知道武器什么时候能打下一轮 |
| **需求** | 在玩家周围显示武器冷却环形进度 |
| **修改位置** | `Weapon.render` 或 `GameManager.renderHUD` |

#### 代码示例：
```javascript
// 在武器基类的 render 中添加
render(ctx, player) {
    // 冷却环半径
    const radius = 25;
    const cx = player.x;
    const cy = player.y;
    
    // 计算冷却进度
    const cooldownRatio = this.timer / this.getEffectiveCooldown();
    
    // 背景灰环
    ctx.beginPath();
    ctx.arc(cx, cy, radius, 0, Math.PI * 2);
    ctx.strokeStyle = 'rgba(100, 100, 100, 0.5)';
    ctx.lineWidth = 3;
    ctx.stroke();
    
    // 前景进度环
    if (cooldownRatio > 0) {
        ctx.beginPath();
        ctx.arc(cx, cy, radius, -Math.PI/2, -Math.PI/2 + Math.PI*2*(1-cooldownRatio));
        ctx.strokeStyle = '#00ff00';  // 绿色表示就绪
        ctx.lineWidth = 3;
        ctx.stroke();
    }
}
```

// GameManager.update 新增摄像机更新
updateCamera() {
    const targetX = this.player.x - this.canvas.width / 2;
    const targetY = this.player.y - this.canvas.height / 2;
    
    // 平滑跟随
    this.camera.x += (targetX - this.camera.x) * this.camera.smoothness;
    this.camera.y += (targetY - this.camera.y) * this.camera.smoothness;
    
    // 边界限制，摄像机不出地图
    this.camera.x = Math.max(0, Math.min(this.mapWidth - this.canvas.width, this.camera.x));
    this.camera.y = Math.max(0, Math.min(this.mapHeight - this.canvas.height, this.camera.y));
}
